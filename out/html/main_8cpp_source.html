279  auto &V = p.velocity;
280  auto &F = p.force;
281  F = glm::vec3(0);
282 
283  if (!p.isFixedPoint) F += gravity;
284 
285  F += DEFAULT_DAMPING * V;
286  }
287 
291  for (auto &s : springs) {
292  auto &P1 = s.p1.pos;
293  auto &V1 = s.p1.velocity;
294  auto &F1 = s.p1.force;
295  auto &P2 = s.p2.pos;
296  auto &V2 = s.p2.velocity;
297  auto &F2 = s.p2.force;
298 
299  glm::vec3 deltaP = P1 - P2;
300  glm::vec3 deltaV = V1 - V2;
301  float dist = glm::length(deltaP);
302 
303  float leftTerm = -s.Ks * (dist - s.restLength);
304  float rightTerm = s.Kd * (glm::dot(deltaV, deltaP) / dist);
305  glm::vec3 springForce = (leftTerm + rightTerm) * glm::normalize(deltaP);

308  if (!s.p2.isFixedPoint) F2 -= springForce;
